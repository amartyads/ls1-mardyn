134inline void RegionParticleIterator :: next_non_empty_cell() {
135 //cellIndex should always be the index in the cell array (_cells member variable in LinkedCells)
136 mardyn_assert(_cells != nullptr);
137
138 const CellContainer_T& cells = *_cells;
139 const CellIndex_T numCellsInRegion = _regionDimensions[2] * _regionDimensions[1] * _regionDimensions[0];
140
141 // find the next non-empty cell
142 for (_localCellIndex += _stride; _localCellIndex < numCellsInRegion; _localCellIndex += _stride) {
143
144 const ParticleCellBase & c = cells.at(getGlobalCellIndex());
145
146 if (c.isNotEmpty() and (_type == ALL_CELLS or not c.isHaloCell())) {
147 _cell_index = getGlobalCellIndex();
148 updateCellIteratorCell();
149 break;
150 }
151 }

154inline ParticleIterator::CellIndex_T RegionParticleIterator :: getGlobalCellIndex() {
155 CellIndex_T dz = _localCellIndex / (_regionDimensions[0] * _regionDimensions[1]);
156 CellIndex_T dy = (_localCellIndex % (_regionDimensions[0] * _regionDimensions[1])) / _regionDimensions[0];
157 CellIndex_T dx = (_localCellIndex % (_regionDimensions[0] * _regionDimensions[1])) % _regionDimensions[0];
158 return (_baseX + dx) + (_baseY + dy) * _globalDimensions[0] + (_baseZ + dz) * _globalDimensions[0] * _globalDimensions[1];
159}
